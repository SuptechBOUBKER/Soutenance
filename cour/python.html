from math import *

class Point:
    def __init__(self,a=0,b=0):

        self.y=b
    def get_x(self):

    def get_y(self):
        return self.y

    def set_x(self,a):
        self.x=a 

    def set_y(self, b):
        self.y=b

        self.set_x(self.get_x()+dx)
        self.set_y(self.get_y()+dy)

        print("x=",self.get_x())
        print("y=",self.get_y())

        print("donner les coordonnées")
        self.x = int(input ("x = "))
        self.y = int(input ("y = "))

        x2=(self.get_y()-p.get_y())*(self.get_y()-p.get_y());
        d=sqrt(x1+x2)
        return d

    def milieu(self, p):
        p1 = Point();
        p1.x=(self.get_x()+p.get_x())/2
        p1.y=(self.get_y()+p.get_y())/2
        return p1

                                  Réponse

#coding: utf-8
class Rectangle:
    def __init__(self,longueur,largeur):
        self.longueur = longueur
        self.largeur = largeur
       
    # Méthode qui calcul le périmètre
    def Perimetre(self):
        return 2*(self.longueur + self.largeur)
   
    # Méthode qui calcul la surface
    def Surface(self):
        return self.longueur*self.largeur
   
class Parallelepipede(Rectangle):
    def __init__(self,longueur,largeur, hauteur):
        Rectangle.__init__(self,longueur,largeur)
        self.hauteur = hauteur
   
    # méthode qui calcul le volume
    def Volume(self):
        return self.longueur*self.largeur*self.hauteur

        return pi*self.r**2
